### T-2: Status transition validation

**File:** `apps/api/src/orders/order-status.transitions.ts`

Define valid transitions:
```typescript
export const VALID_TRANSITIONS: Record<string, string[]> = {
  pending: ['ordered', 'cancelled', 'failed'],
  ordered: ['processing', 'cancelled', 'failed'],
  processing: ['shipped', 'cancelled', 'failed'],
  shipped: ['delivered', 'failed'],
  delivered: [],  // terminal
  cancelled: [],  // terminal
  failed: [],     // terminal
};

export function isValidTransition(from: string, to: string): boolean {
  return VALID_TRANSITIONS[from]?.includes(to) ?? false;
}
```

Integrate into a `transitionStatus()` method on OrdersService that validates the transition, updates the order, records history, and updates tracking fields.

### T-4: HMAC signature verification utility

**File:** `apps/api/src/orders/webhooks/verify-signature.ts`

```typescript
import { createHmac, timingSafeEqual } from 'crypto';

export function verifyWebhookSignature(payload: Buffer, signature: string, secret: string): boolean {
  const expected = createHmac('sha256', secret).update(payload).digest('hex');
  return timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```
